\NewDocumentCommand{\thmProjectProofStart}{m}{%
    \vspace{-10pt}%
    \IfStrEq{#1}{\proofname}{%
        \begin{replacementproof}%
    }{%
        \begin{replacementproof}[#1]%
    }%
}
\NewDocumentCommand{\thmProjectProofStop}{}{\end{replacementproof}}

    \vspace{-10pt}%
    \IfBlankTF{#1}{\begin{tmpexplanation}}{\begin{tmpexplanation}[#1]}%
}
\NewDocumentCommand{\thmProjectExplanationStop}{}{\end{tmpexplanation}}
